113#[derive(Debug, Clone, Hash)]
114pub struct Polygon<T> {
115  // Use points: Arc<Vec<Point<T, 2>>>, ?
116  // Key: PointId
117  pub(crate) points: Vec<Point<T>>,
118  // Key: PointId
119  pub(crate) ring_index: Vec<RingId>,
120  // Key: PointId
121  pub(crate) position_index: Vec<PositionId>,
122  // Ring 0 is boundary and ccw

129impl<T> Polygon<T> {
130  /// $O(1)$
131  pub fn new_unchecked(vertices: Vec<Point<T, 2>>) -> Polygon<T>
132  where
133    T: PolygonScalar,
134  {
135    let len = vertices.len();
136    Polygon {
137      points: vertices,
138      ring_index: vec![RingId(0); len],
139      position_index: (0..len).map(PositionId).collect(),
140      rings: vec![(0..len).map(PointId).collect()],
141    }
142  }
143
144  /// $O(n^2)$
145  pub fn new(points: Vec<Point<T, 2>>) -> Result<Polygon<T>, Error>
146  where
147    T: PolygonScalar,
148  {
149    // Verify minimum length before calling 'ensure_ccw'.
150    if points.len() < 3 {
151      return Err(Error::InsufficientVertices);
152    }
153    let mut p = Self::new_unchecked(points);
154    p.ensure_ccw()?;
155    p.validate()?;
156    Ok(p)
157  }

159  // Validate that a polygon is simple.
160  // https://en.wikipedia.org/wiki/Simple_polygon
161  pub fn validate(&self) -> Result<(), Error>
162  where
163    T: PolygonScalar,
164  {
165    // Has no duplicate points.
166    let mut seen = std::collections::BTreeSet::new();
167    for pt in self.iter() {
168      if !seen.insert(pt) {
169        return Err(Error::DuplicatePoints);
170      }
171    }
172
173    self.validate_weakly()
174  }
175
176  pub fn validate_weakly(&self) -> Result<(), Error>
177  where
178    T: PolygonScalar,
179  {
180    // Rings is never allowed to be empty under any circumstances, even when
181    // using 'Polygon::unchecked_new'.
182    assert!(!self.rings.is_empty());
183
184    // Has at least three points.
185    if self.rings[0].len() < 3 {
186      return Err(Error::InsufficientVertices);
187    }
188    // Is counter-clockwise
189    if self.orientation() != Orientation::CounterClockWise {
190      return Err(Error::ClockWiseViolation);
191    }
192    // Has no self intersections.
193    // XXX: Hm, allow overlapping (but not crossing) edges in the weakly check?
194    let edges: Vec<DirectedEdge<'_, T, 2>> = self.iter_boundary_edges().collect();
195    let mut isects = crate::algorithms::segment_intersections(&edges);
196    if isects.next().is_some() {
197      return Err(Error::SelfIntersections);
198    }
199    Ok(())
200  }

202  pub fn locate(&self, origin: &Point<T, 2>) -> PointLocation
203  where
204    T: PolygonScalar,
205  {
206    // FIXME: Support polygons with holes.
207    assert_eq!(
208      self.rings.len(),
209      1,
210      "FIXME: Polygon::locate should support polygons with holes."
211    );
212
213    let direction = Vector::unit_right();
214    let ray = HalfLineSoS::new_directed(origin, &direction);
215    let mut intersections = 0;
216    for edge in self.iter_boundary_edges() {
217      if edge.contains(origin) {
218        return PointLocation::OnBoundary;
219      }
220      if let Some(Crossing(lean)) = ray.intersect(edge) {
221        // Only count crossing that aren't leaning to the right.
222        if !lean.is_cw() {
223          intersections += 1;
224        }
225      }
226    }
227    if intersections % 2 == 0 {
228      PointLocation::Outside
229    } else {
230      PointLocation::Inside
231    }
232  }

244  //
245  // # Panics
246  //
247  // May panic for bounded types (i8, isize, etc). Maybe produce inaccurate results for
248  // floating point types (f32, f64).
249  pub fn centroid(&self) -> Point<T>
250  where
251    T: PolygonScalar,
252  {
253    let xs: Vector<T, 2> = self
254      .iter_boundary_edges()
255      .map(|edge| {
256        let p = edge.src.as_vec();
257        let q = edge.dst.as_vec();
258        (p + q) * (p.0[0].clone() * q.0[1].clone() - q.0[0].clone() * p.0[1].clone())
259      })
260      .sum();
261    let three = T::from_constant(3);
262    Point::from(xs / (three * self.signed_area_2x()))
263  }

265  pub fn bounding_box(&self) -> (Point<T>, Point<T>)
266  where
267    T: PolygonScalar,
268  {
269    assert!(!self.rings[0].is_empty());
270    let mut min_x = self.points[self.rings[0][0].usize()].x_coord().clone();
271    let mut max_x = self.points[self.rings[0][0].usize()].x_coord().clone();
272    let mut min_y = self.points[self.rings[0][0].usize()].y_coord().clone();
273    let mut max_y = self.points[self.rings[0][0].usize()].y_coord().clone();
274    for pt in self.iter_boundary() {
275      if pt.point().x_coord() < &min_x {
276        min_x = pt.point().x_coord().clone();
277      }
278      if pt.point().x_coord() > &max_x {
279        max_x = pt.point().x_coord().clone();
280      }
281      if pt.point().y_coord() < &min_y {
282        min_y = pt.point().y_coord().clone();
283      }
284      if pt.point().y_coord() > &max_y {
285        max_y = pt.point().y_coord().clone();
286      }
287    }
288    (Point::new([min_x, min_y]), Point::new([max_x, max_y]))
289  }

291  /// Computes the area of a polygon. If the polygon winds counter-clockwise,
292  /// the area will be a positive number. If the polygon winds clockwise, the area will
293  /// be negative.
294  ///
295  /// # Return type
296  ///
297  /// This function is polymorphic for the same reason as [`signed_area_2x`](Polygon::signed_area_2x).
298  ///
299  /// # Time complexity
300  ///
301  /// $O(n)$
302  ///
303  /// # Examples
304  /// ```rust
305  /// // The area of this polygon is 1/2 and won't fit in an `i32`
306  /// # use rgeometry::data::*;
307  /// # fn main() -> Result<(), rgeometry::Error> {
308  /// let p = Polygon::new(vec![
309  ///   Point::new([0, 0]),
310  ///   Point::new([1, 0]),
311  ///   Point::new([0, 1]),
312  ///  ])?;
313  /// assert_eq!(p.signed_area::<i32>(), 0);
314  /// # Ok(())
315  /// # }
316  /// ```
317  ///
318  /// ```rust
319  /// // The area of this polygon is 1/2 and will fit in a `Ratio<i32>`.
320  /// # use rgeometry::data::*;
321  /// # use num_rational::Ratio;
322  /// # fn main() -> Result<(), rgeometry::Error> {
323  /// let p = Polygon::new(vec![
324  ///   Point::new([0, 0]),
325  ///   Point::new([1, 0]),
326  ///   Point::new([0, 1]),
327  ///  ])?;
328  /// assert_eq!(p.signed_area::<Ratio<i32>>(), Ratio::from((1,2)));
329  /// # Ok(())
330  /// # }
331  /// ```
332  pub fn signed_area<F>(&self) -> F
333  where
334    T: PolygonScalar + Into<F>,
335    F: NumOps<F, F> + Sum + FromPrimitive,
336  {
337    self.signed_area_2x::<F>() / F::from_usize(2).unwrap()
338  }

340  /// Compute double the area of a polygon. If the polygon winds counter-clockwise,
341  /// the area will be a positive number. If the polygon winds clockwise, the area will
342  /// be negative.
343  ///
344  /// Why compute double the area? If you are using integer coordinates then the doubled
345  /// area will also be an integer value. Computing the exact requires a division which
346  /// may leave you with a non-integer result.
347  ///
348  /// # Return type
349  ///
350  /// Storing the area of a polygon may require more bits of precision than are used
351  /// for the coordinates. For example, if 32bit integers are used for point coordinates
352  /// then you might need 65 bits to store the area doubled. For this reason, the return
353  /// type is polymorphic and should be selected with care. If you are unsure which type
354  /// is appropriate, use [`BigInt`](num::BigInt) or [`BigRational`](num::BigRational).
355  ///
356  /// # Time complexity
357  ///
358  /// $O(n)$
359  ///
360  /// # Examples:
361  /// ```rust
362  /// // The area of this polygon is 1/2 and cannot fit in an `i32` variable
363  /// // so lets compute twice the area.
364  /// # use rgeometry::data::*;
365  /// # fn main() -> Result<(), rgeometry::Error> {
366  /// let p = Polygon::new(vec![
367  ///   Point::new([0, 0]),
368  ///   Point::new([1, 0]),
369  ///   Point::new([0, 1]),
370  ///  ])?;
371  /// assert_eq!(p.signed_area_2x::<i32>(), 1);
372  /// # Ok(())
373  /// # }
374  /// ```
375  ///
376  /// ```rust
377  /// // The area of a polygon may require more bits of precision than are
378  /// // used for the coordinates.
379  /// # use num::BigInt;
380  /// # use rgeometry::data::*;
381  /// # fn main() -> Result<(), rgeometry::Error> {
382  /// let p = Polygon::new(vec![
383  ///   Point::new([0, 0]),
384  ///   Point::new([i32::MAX, 0]),
385  ///   Point::new([0, i32::MAX]),
386  ///  ])?;
387  /// assert_eq!(p.signed_area_2x::<i64>(), 4611686014132420609_i64);
388  /// # Ok(())
389  /// # }
390  /// ```
391  pub fn signed_area_2x<F>(&self) -> F
392  where
393    T: PolygonScalar + Into<F>,
394    F: NumOps<F, F> + Sum,
395  {
396    self
397      .iter_boundary_edges()
398      .map(|edge| {
399        let p = edge.src;
400        let q = edge.dst;
401        p.array[0].clone().into() * q.array[1].clone().into()
402          - q.array[0].clone().into() * p.array[1].clone().into()
403      })
404      .sum()
405  }

445  pub fn equals(&self, other: &Self) -> bool
446  where
447    T: PolygonScalar,
448  {
449    if self.boundary_slice().len() != other.boundary_slice().len() {
450      return false;
451    }
452    for pos in self.iter_boundary() {
453      let iter = CursorIter {
454        cursor_head: pos,
455        cursor_tail: pos.prev(),
456        exhausted: false,
457      };
458      if iter == other.iter_boundary() {
459        return true;
460      }
461    }
462    false
463  }

546  // Reverse all points between p1 and p2, inclusive of p1 and p2.
547  // For example: p1, a, b, c, p2 =>
548  //              p2, c, b, a, p1
549  pub fn vertices_reverse(&mut self, mut pt1: Position, mut pt2: Position) {
550    loop {
551      self.swap_positions(pt1, pt2);
552      pt1.move_next();
553      if pt1 == pt2 {
554        break;
555      }
556      pt2.move_prev();
557      if pt1 == pt2 {
558        break;
559      }
560    }
561  }

563  // Move pt2 back until it is right in front of p1.
564  pub fn vertices_join(&mut self, pt1: Position, mut pt2: Position) {
565    while pt2.prev() != pt1 {
566      let prev = pt2.prev();
567      self.swap_positions(prev, pt2);
568      pt2.position_id = prev.position_id;
569    }
570  }

572  /// Panics if the edge isn't part of the polygon.
573  pub fn direct(&self, edge: IndexEdge) -> DirectedIndexEdge {
574    let min_cursor = self.cursor(edge.min);
575    let max_cursor = self.cursor(edge.max);
576    if min_cursor.next() == max_cursor {
577      DirectedIndexEdge {
578        src: edge.min,
579        dst: edge.max,
580      }
581    } else if max_cursor.next() == min_cursor {
582      DirectedIndexEdge {
583        src: edge.max,
584        dst: edge.min,
585      }
586    } else {
587      panic!("Edge isn't part of polygon: {:?}", edge)
588    }
589  }

591  pub fn ensure_ccw(&mut self) -> Result<(), Error>
592  where
593    T: PolygonScalar,
594  {
595    match self.orientation() {
596      Orientation::CounterClockWise => Ok(()),
597      Orientation::CoLinear => Err(Error::CoLinearViolation),
598      Orientation::ClockWise => {
599        let root_position = Position {
600          ring_id: RingId(0),
601          position_id: PositionId(0),
602          size: self.rings[0].len(),
603        };
604        self.vertices_reverse(root_position, root_position.prev());
605        Ok(())
606      }
607    }
608  }

614  fn swap_positions(&mut self, pa: Position, pb: Position) {
615    assert_eq!(pa.ring_id, pb.ring_id);
616
617    let ring = &mut self.rings[pa.ring_id.0];
618
619    let pa_point_id: PointId = ring[pa.position_id.0];
620    let pb_point_id: PointId = ring[pb.position_id.0];
621    ring.swap(pa.position_id.0, pb.position_id.0);
622    self.position_index.swap(pa_point_id.0, pb_point_id.0);
623  }

633impl Polygon<OrderedFloat<f64>> {
634  #[must_use]
635  // Center on <0,0>. Scale size such that max(width,height) = 1.
636  pub fn normalize(&self) -> Polygon<OrderedFloat<f64>> {
637    let (min, max) = self.bounding_box();
638    let [min_x, min_y] = min.array;
639    let [max_x, max_y] = max.array;
640    let width = max_x - min_x;
641    let height = max_y - min_y;
642    let ratio = std::cmp::max(width, height);
643    let centroid = self.centroid();
644    let t = Transform::translate(-Vector::from(centroid));
645    let s = Transform::uniform_scale(ratio.recip());
646    s * t * self
647  }
648}

714  }

720  // O(n)
721  pub fn is_ear(&self) -> bool
722  where
723    T: PolygonScalar,
724  {
725    let trig =
726      TriangleView::new_unchecked([self.prev().point(), self.point(), self.next().point()]);
727    if trig.orientation() == Orientation::CounterClockWise {
728      for pt in self.next().next().to(Excluded(self.prev())) {
729        if trig.locate(pt.point()) != PointLocation::Outside {
730          return false;
731        }
732      }
733      true
734    } else {
735      false
736    }
737  }

739  pub fn orientation(&self) -> Orientation
740  where
741    T: PolygonScalar,
742  {
743    let p1 = self.prev().point();
744    let p2 = self.point();
745    let p3 = self.next().point();
746    Point::orient(p1, p2, p3)
747  }

749  pub fn is_colinear(&self) -> bool
750  where
751    T: PolygonScalar,
752  {
753    self.orientation() == Orientation::CoLinear
754  }

756  pub fn to(self: Cursor<'a, T>, end: Bound<Cursor<'a, T>>) -> CursorIter<'a, T> {
757    // FIXME: Check that self and end point to the same polygon.
758    match end {
759      Bound::Unbounded => CursorIter {
760        cursor_head: self,
761        cursor_tail: self.prev(),
762        exhausted: false,
763      },
764      Bound::Included(end) => CursorIter {
765        cursor_head: self,
766        cursor_tail: end,
767        exhausted: false,
768      },
769      Bound::Excluded(end) => CursorIter {
770        cursor_head: self,
771        cursor_tail: end.prev(),
772        exhausted: false,
773      },
774    }
775  }

778#[derive(Debug, Copy, Clone, PartialEq, Eq, Ord, PartialOrd)]
779pub struct Position {
780  ring_id: RingId,         // Ring identifier
781  size: usize,             // Ring size
782  position_id: PositionId, // Position in ring. 0 <= position_id < size
783}
784
785impl Position {
786  #[must_use]
787  pub fn prev(mut self) -> Position {
788    self.move_prev();
789    self
790  }
791
792  #[must_use]
793  pub fn next(mut self) -> Position {
794    self.move_next();
795    self
796  }
797
798  pub fn move_next(&mut self) {
799    if self.position_id.0 == self.size - 1 {
800      self.position_id.0 = 0;
801    } else {
802      self.position_id.0 += 1;
803    }
804  }
805
806  pub fn move_prev(&mut self) {
807    if self.position_id.0 == 0 {
808      self.position_id.0 = self.size - 1;
809    } else {
810      self.position_id.0 -= 1;
811    }
812  }
813}

985  // Locate a point relative to a polygon. Should be identical to
986  // Polygon::locate but slower.
987  fn locate_by_triangulation<T>(poly: &Polygon<T>, origin: &Point<T>) -> PointLocation
988  where
989    T: PolygonScalar,
990  {
991    for edge in poly.iter_boundary_edges() {
992      if edge.contains(origin) {
993        return PointLocation::OnBoundary;
994      }
995    }
996    for (a, b, c) in poly.triangulate() {
997      let trig = TriangleView::new([&a, &b, &c]).expect("valid triangle");
998      if trig.locate(origin) != PointLocation::Outside {
999        return PointLocation::Inside;
1000      }
1001    }
1002    PointLocation::Outside
1003  }
