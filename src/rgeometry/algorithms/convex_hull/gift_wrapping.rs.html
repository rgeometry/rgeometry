9//    No panics.

12/// Convex hull of a set of points.
13///
14/// [Gift Wrapping][wiki] algorithm for finding the smallest convex polygon which
15/// contains all the given points.
16///
17/// # Errors
18/// Will return an error iff the input set contains less than three distinct points.
19///
20/// # Properties
21/// * No points from the input set will be outside the returned convex polygon.
22/// * All vertices in the convex polygon are from the input set.
23///
24/// # Time complexity
25/// $O(n \log h)$ as h is the size of the points on convex hull
26///
27/// # Examples
28///
29/// ```rust
30/// # pub fn main() {
31/// # use rgeometry::algorithms::convex_hull;
32/// # use rgeometry::data::Point;
33/// # use rgeometry::Error;
34/// let empty_set: Vec<Point<i32,2>> = vec![];
35/// assert_eq!(
36///   convex_hull(empty_set).err(),
37///   Some(Error::InsufficientVertices))
38/// # }
39/// ```
40///
41/// ```rust
42/// # pub fn main() {
43/// # use rgeometry::algorithms::convex_hull;
44/// # use rgeometry::data::Point;
45/// # use rgeometry::Error;
46/// let dups = vec![Point::new([0,0])].repeat(3);
47/// assert_eq!(
48///   convex_hull(dups).err(),
49///   Some(Error::InsufficientVertices))
50/// # }
51/// ```
52///
53/// [wiki]: https://en.wikipedia.org/wiki/Gift_wrapping_algorithm
54// ```no_run
55// # pub fn main() {
56// #   use rgeometry::algorithms::convex_hull;
57// #   use rgeometry_wasm::playground::*;
58// #
59// #   static START: std::sync::Once = std::sync::Once::new();
60// #   START.call_once(|| on_mousemove(|_event| main()));
61// #
62// #   clear_screen();

70// #   context().fill();

73// #     render_point(&pt);

80pub fn convex_hull<T>(pts: Vec<Point<T>>) -> Result<PolygonConvex<T>, Error>
81where
82  T: PolygonScalar,
83{
84  let n = pts.len();
85  if n < 3 {
86    return Err(Error::InsufficientVertices);
87  }
88  let leftmost = leftmost_point_index(&pts)?;
89
90  let mut hull: Vec<Point<T>> = Vec::with_capacity(pts.len());
91  let mut p = leftmost;
92
93  loop {
94    hull.push(pts[p].clone());
95    let mut q = (p + 1) % n;
96
97    for i in 0..n {
98      let orientation = Point::orient(&pts[p], &pts[i], &pts[q]);
99      // check if 3 points are coliner, as we want to add the minimal number of points on convex hull
100      if orientation == Orientation::CounterClockWise
101        || (orientation == Orientation::CoLinear
102          && pts[p].cmp_distance_to(&pts[i], &pts[q]) == Ordering::Greater)
103      {
104        q = i;
105      }
106    }
107
108    p = q;
109    if p == leftmost {
110      break;
111    }
112  }
113
114  if hull.len() < 3 {
115    return Err(Error::InsufficientVertices);
116  }
117
118  Ok(PolygonConvex::new_unchecked(Polygon::new_unchecked(hull)))
119}
120
121// Finds the smallest point index
122// O(n)
123fn leftmost_point_index<T>(pts: &[Point<T>]) -> Result<usize, Error>
124where
125  T: PolygonScalar,
126{
127  pts
128    .iter()
129    .enumerate()
130    .min_by(|(_, a), (_, b)| {
131      TotalOrd::total_cmp(&(a.y_coord(), a.x_coord()), &(b.y_coord(), b.x_coord()))
132    })
133    .map(|(index, _)| index)
134    .ok_or(Error::InsufficientVertices)
135}

245  #[proptest]
246  fn convex_hull_prop(#[strategy(vec(any_r(), 0..100))] pts: Vec<Point<BigInt>>) {
247    if let Ok(poly) = convex_hull(pts.clone()) {
248      // Prop #1: Results are valid.
249      prop_assert_eq!(poly.validate().err(), None);
250      // Prop #2: No points from the input set are outside the polygon.
251      for pt in pts.iter() {
252        prop_assert_ne!(poly.locate(pt), PointLocation::Outside)
253      }
254      // Prop #3: All vertices are in the input set.
255      for pt in poly.iter() {
256        prop_assert!(pts.contains(pt))
257      }
258    }
259  }
