7//! # Computational Geometry
8//!
9//! RGeometry provides data types (points, polygons, lines, segments) and algorithms for
10//! computational geometry. Computational geometry is the study of algorithms for solving
11//! geometric problems, such as computing convex hulls, triangulating polygons, finding
12//! intersections, and determining visibility.
13//!
14//! This crate includes algorithms for:
15//! - **Convex hulls**: Graham scan, gift wrapping, and Melkman's algorithm
16//! - **Triangulation**: Ear clipping for simple polygons and constrained Delaunay triangulation
17//! - **Polygonization**: Converting point sets to polygons (monotone, star-shaped, two-opt)
18//! - **Line segment intersection**: Naive and Bentley-Ottmann sweep line algorithms
19//! - **Visibility**: Computing visibility polygons
20//! - **Spatial indexing**: Z-order curve hashing
21//!
22//! # Demos
23//!
24//! Interactive visualizations are available:
25//!
26//! - [Convex Hull](../../../convex_hull.html) - Graham scan algorithm
27//! - [Melkman's Algorithm](../../../melkman.html) - Online convex hull for simple polylines
28//! - [Ear Clipping](../../../earclip.html) - Polygon triangulation
29//! - [Constrained Delaunay](../../../delaunay.html) - Ear-clipping plus Lawson flips
30//! - [Two-Opt](../../../two_opt.html) - Polygonization heuristic
31//! - [Random Convex](../../../random_convex.html) - Convex polygon generation
32//! - [Random Monotone](../../../random_monotone.html) - Monotone polygon generation
33//!
34//! # Caveats
35//!
36//! RGeometry supports multiple numeric types, each with different trade-offs:
37//!
38//! - **Arbitrary precision** (e.g., `num::BigRational`): Always produces mathematically correct
39//!   results but can be significantly slower due to dynamic memory allocation and arbitrary
40//!   precision arithmetic.
41//!
42//! - **Fixed precision integers** (e.g., `i32`, `i64`): RGeometry guarantees that intermediate
43//!   calculations will not overflow or underflow, but final outputs may not be representable in
44//!   the chosen type. For example, testing if two line segments overlap is always safe, but
45//!   computing where they intersect may produce a point whose coordinates cannot be represented.
46//!
47//! - **Floating point** (e.g., `f32`, `f64`): RGeometry uses robust geometric predicates to
48//!   ensure accumulated rounding errors do not affect algorithmic results. However, individual
49//!   coordinate values are still subject to floating-point precision limits.

69pub enum Error {
70  InsufficientVertices,
71  SelfIntersections,
72  DuplicatePoints,
73  /// Two consecutive line segments are either colinear or oriented clockwise.
74  ConvexViolation,
75  ClockWiseViolation,
76  CoLinearViolation,
77}

79impl std::fmt::Display for Error {
80  fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> Result<(), std::fmt::Error> {
81    match self {
82      Error::InsufficientVertices => write!(f, "Insufficient vertices"),
83      Error::SelfIntersections => write!(f, "Self intersections"),
84      Error::DuplicatePoints => write!(f, "Duplicate points"),
85      Error::ConvexViolation => write!(f, "Convex violation"),
86      Error::ClockWiseViolation => write!(f, "Clockwise violation"),
87      Error::CoLinearViolation => write!(
88        f,
89        "Two or more points are colinear and no valid solution exists"
90      ),
91    }
92  }
93}

95pub trait TotalOrd {
96  fn total_cmp(&self, other: &Self) -> Ordering;
97
98  fn total_min(self, other: Self) -> Self
99  where
100    Self: Sized,
101  {
102    std::cmp::min_by(self, other, TotalOrd::total_cmp)
103  }
104
105  fn total_max(self, other: Self) -> Self
106  where
107    Self: Sized,
108  {
109    std::cmp::max_by(self, other, TotalOrd::total_cmp)
110  }
111}
112
113impl TotalOrd for u32 {
114  fn total_cmp(&self, other: &Self) -> Ordering {
115    self.cmp(other)
116  }
117}
118
119impl<A: TotalOrd> TotalOrd for &A {
120  fn total_cmp(&self, other: &Self) -> Ordering {
121    (*self).total_cmp(*other)
122  }
123}
124
125impl<A: TotalOrd, B: TotalOrd> TotalOrd for (A, B) {
126  fn total_cmp(&self, other: &Self) -> Ordering {
127    self
128      .0
129      .total_cmp(&other.0)
130      .then_with(|| self.1.total_cmp(&other.1))
131  }
132}

161fn approx_intersection_point_basic<T: Clone + NumOps<T, T> + PartialEq + std::fmt::Debug>(
162  zero: T,
163  p1: [T; 2],
164  p2: [T; 2],
165  q1: [T; 2],
166  q2: [T; 2],
167) -> Option<[T; 2]> {
168  let [x1, y1] = p1;
169  let [x2, y2] = p2;
170  let [x3, y3] = q1;
171  let [x4, y4] = q2;
172  let denom: T = (x1.clone() - x2.clone()) * (y3.clone() - y4.clone())
173    - (y1.clone() - y2.clone()) * (x3.clone() - x4.clone());
174  if denom == zero {
175    return None;
176  }
177  let part_a = x1.clone() * y2.clone() - y1.clone() * x2.clone();
178  let part_b = x3.clone() * y4.clone() - y3.clone() * x4.clone();
179  let x_num =
180    part_a.clone() * (x3.clone() - x4.clone()) - (x1.clone() - x2.clone()) * part_b.clone();
181  let y_num = part_a * (y3.clone() - y4.clone()) - (y1.clone() - y2.clone()) * part_b;
182  Some([x_num / denom.clone(), y_num / denom])
183}

185fn incircle_exact<T>(a: &[T; 2], b: &[T; 2], c: &[T; 2], d: &[T; 2]) -> Ordering
186where
187  T: Clone + NumOps<T, T> + Zero + Ord,
188{
189  let ax = a[0].clone() - d[0].clone();
190  let ay = a[1].clone() - d[1].clone();
191  let bx = b[0].clone() - d[0].clone();
192  let by = b[1].clone() - d[1].clone();
193  let cx = c[0].clone() - d[0].clone();
194  let cy = c[1].clone() - d[1].clone();
195
196  let a_len = ax.clone() * ax.clone() + ay.clone() * ay.clone();
197  let b_len = bx.clone() * bx.clone() + by.clone() * by.clone();
198  let c_len = cx.clone() * cx.clone() + cy.clone() * cy.clone();
199
200  let det = det3_generic(&ax, &ay, &a_len, &bx, &by, &b_len, &cx, &cy, &c_len);
201  det.cmp(&T::zero())
202}

204#[allow(clippy::too_many_arguments)]
205fn det3_generic<T>(a1: &T, a2: &T, a3: &T, b1: &T, b2: &T, b3: &T, c1: &T, c2: &T, c3: &T) -> T
206where
207  T: Clone + NumOps<T, T>,
208{
209  let m1 = b2.clone() * c3.clone() - b3.clone() * c2.clone();
210  let m2 = b1.clone() * c3.clone() - b3.clone() * c1.clone();
211  let m3 = b1.clone() * c2.clone() - b2.clone() * c1.clone();
212  a1.clone() * m1 - a2.clone() * m2 + a3.clone() * m3
213}

215macro_rules! fixed_precision {
216  ( $ty:ty, $uty:ty, $long:ty, $ulong: ty, $apfp_mod:path ) => {
217    impl TotalOrd for $ty {
218      fn total_cmp(&self, other: &Self) -> Ordering {
219        self.cmp(other)
220      }
221    }
222
223    impl PolygonScalar for $ty {
224      fn from_constant(val: i8) -> Self {
225        val as $ty
226      }
227      fn cmp_dist(p: &[Self; 2], q: &[Self; 2], r: &[Self; 2]) -> std::cmp::Ordering {
228        use $apfp_mod as apfp_mod;
229        apfp_mod::cmp_dist(
230          &apfp_mod::Coord::new(p[0], p[1]),
231          &apfp_mod::Coord::new(q[0], q[1]),
232          &apfp_mod::Coord::new(r[0], r[1]),
233        )
234      }
235
236      fn cmp_slope(p: &[Self; 2], q: &[Self; 2], r: &[Self; 2]) -> std::cmp::Ordering {
237        use apfp::geometry::Orientation;
238        use $apfp_mod as apfp_mod;
239        let orient = apfp_mod::orient2d(
240          &apfp_mod::Coord::new(p[0], p[1]),
241          &apfp_mod::Coord::new(q[0], q[1]),
242          &apfp_mod::Coord::new(r[0], r[1]),
243        );
244        match orient {
245          Orientation::CounterClockwise => Ordering::Greater,
246          Orientation::Clockwise => Ordering::Less,
247          Orientation::CoLinear => Ordering::Equal,
248        }
249      }
250
251      fn cmp_vector_slope(vector: &[Self; 2], p: &[Self; 2], q: &[Self; 2]) -> std::cmp::Ordering {
252        use apfp::geometry::Orientation;
253        use $apfp_mod as apfp_mod;
254        let orient = apfp_mod::orient2d_vec(
255          &apfp_mod::Coord::new(p[0], p[1]),
256          &apfp_mod::Coord::new(vector[0], vector[1]),
257          &apfp_mod::Coord::new(q[0], q[1]),
258        );
259        match orient {
260          Orientation::CounterClockwise => Ordering::Greater,
261          Orientation::Clockwise => Ordering::Less,
262          Orientation::CoLinear => Ordering::Equal,
263        }
264      }
265
266      fn cmp_perp_vector_slope(
267        vector: &[Self; 2],
268        p: &[Self; 2],
269        q: &[Self; 2],
270      ) -> std::cmp::Ordering {
271        use apfp::geometry::Orientation;
272        use $apfp_mod as apfp_mod;
273        let orient = apfp_mod::orient2d_normal(
274          &apfp_mod::Coord::new(p[0], p[1]),
275          &apfp_mod::Coord::new(vector[0], vector[1]),
276          &apfp_mod::Coord::new(q[0], q[1]),
277        );
278        match orient {
279          Orientation::CounterClockwise => Ordering::Greater,
280          Orientation::Clockwise => Ordering::Less,
281          Orientation::CoLinear => Ordering::Equal,
282        }
283      }
284      fn approx_intersection_point(
285        p1: [Self; 2],
286        p2: [Self; 2],
287        q1: [Self; 2],
288        q2: [Self; 2],
289      ) -> Option<[Self; 2]> {
290        let long_p1: [$long; 2] = [p1[0] as $long, p1[1] as $long];
291        let long_p2: [$long; 2] = [p2[0] as $long, p2[1] as $long];
292        let long_q1: [$long; 2] = [q1[0] as $long, q1[1] as $long];
293        let long_q2: [$long; 2] = [q2[0] as $long, q2[1] as $long];
294        let long_result = approx_intersection_point_basic(0, long_p1, long_p2, long_q1, long_q2);
295        long_result.and_then(|[x, y]| Some([Self::try_from(x).ok()?, Self::try_from(y).ok()?]))
296      }
297      fn incircle(a: &[Self; 2], b: &[Self; 2], c: &[Self; 2], d: &[Self; 2]) -> Ordering {
298        use apfp::geometry::Orientation;
299        use $apfp_mod as apfp_mod;
300        let result = apfp_mod::incircle(
301          &apfp_mod::Coord::new(a[0], a[1]),
302          &apfp_mod::Coord::new(b[0], b[1]),
303          &apfp_mod::Coord::new(c[0], c[1]),
304          &apfp_mod::Coord::new(d[0], d[1]),
305        );
306        match result {
307          Orientation::CounterClockwise => Ordering::Greater,
308          Orientation::Clockwise => Ordering::Less,
309          Orientation::CoLinear => Ordering::Equal,
310        }
311      }
312    }
313  };
314}

316macro_rules! arbitrary_precision {
317  ( $( $ty:ty ),* ) => {
318    $(
319      impl TotalOrd for $ty {
320        fn total_cmp(&self, other: &Self) -> Ordering {
321          self.cmp(other)
322        }
323      }
324
325      impl PolygonScalar for $ty {
326      fn from_constant(val: i8) -> Self {
327        <$ty>::from_i8(val).unwrap()
328      }
329      fn cmp_dist(p: &[Self; 2], q: &[Self; 2], r: &[Self; 2]) -> std::cmp::Ordering {
330        let pq_x = &p[0] - &q[0];
331        let pq_y = &p[1] - &q[1];
332        let pq_dist_squared: Self = &pq_x*&pq_x + &pq_y*&pq_y;
333        let pr_x = &p[0] - &r[0];
334        let pr_y = &p[1] - &r[1];
335        let pr_dist_squared: Self = &pr_x*&pr_x + &pr_y*&pr_y;
336        pq_dist_squared.cmp(&pr_dist_squared)
337      }
338
339      fn cmp_slope(p: &[Self; 2], q: &[Self; 2], r: &[Self; 2]) -> std::cmp::Ordering {
340        let slope1 = (&r[1] - &q[1]) * (&q[0] - &p[0]);
341        let slope2 = (&q[1] - &p[1]) * (&r[0] - &q[0]);
342        slope1.cmp(&slope2)
343      }
344      fn cmp_vector_slope(vector: &[Self;2], p: &[Self; 2], q: &[Self; 2]) -> std::cmp::Ordering {
345        PolygonScalar::cmp_slope(
346          p,
347          &[&p[0] + &vector[0], &p[1] + &vector[1]],
348          q
349        )
350      }
351      fn cmp_perp_vector_slope(vector: &[Self;2], p: &[Self; 2], q: &[Self; 2]) -> std::cmp::Ordering {
352        PolygonScalar::cmp_slope(
353          p,
354          &[&p[0] - &vector[1], &p[1] + &vector[0]],
355          q
356        )
357      }
358      fn incircle(
359        a: &[Self; 2],
360        b: &[Self; 2],
361        c: &[Self; 2],
362        d: &[Self; 2],
363      ) -> Ordering {
364        incircle_exact(a, b, c, d)
365      }
366    })*
367  };
368}

370macro_rules! floating_precision {
371  ( $( $ty:ty ),* ) => {
372    $(
373      impl TotalOrd for $ty {
374        fn total_cmp(&self, other: &Self) -> Ordering {
375          // The ordering established by `<$ty>::total_cmp` does not always
376          // agree with the PartialOrd and PartialEq implementations of <$ty>. For
377          // example, they consider negative and positive zero equal, while
378          // total_cmp doesnâ€™t.
379          if *self == 0.0 && *other == 0.0 {
380            Ordering::Equal
381          } else {
382            <$ty>::total_cmp(self, other)
383          }
384        }
385      }
386
387      impl PolygonScalar for $ty {
388      fn from_constant(val: i8) -> Self {
389        <$ty>::from_i8(val).unwrap()
390      }
391      // FIXME: Use `geometry_predicates` to speed up calculation. Right now we're
392      // roughly 100x slower than necessary.
393      fn cmp_dist(p: &[Self; 2], q: &[Self; 2], r: &[Self; 2]) -> std::cmp::Ordering {
394        PolygonScalar::cmp_dist(
395          &[float_to_rational(p[0]), float_to_rational(p[1])],
396          &[float_to_rational(q[0]), float_to_rational(q[1])],
397          &[float_to_rational(r[0]), float_to_rational(r[1])],
398        )
399      }
400
401      // This function uses the arbitrary precision machinery of `apfp` to
402      // quickly compute the orientation of three 2D points.
403      fn cmp_slope(p: &[Self; 2], q: &[Self; 2], r: &[Self; 2]) -> std::cmp::Ordering {
404        use apfp::geometry::{Orientation, f64::Coord};
405        let orient = apfp::geometry::f64::orient2d(
406          &Coord::new(p[0] as f64, p[1] as f64),
407          &Coord::new(q[0] as f64, q[1] as f64),
408          &Coord::new(r[0] as f64, r[1] as f64),
409        );
410        match orient {
411          Orientation::CounterClockwise => Ordering::Greater,
412          Orientation::Clockwise => Ordering::Less,
413          Orientation::CoLinear => Ordering::Equal,
414        }
415      }
416      // Uses apfp's orient2d_vec for fast arbitrary precision calculation.
417      fn cmp_vector_slope(vector: &[Self;2], p: &[Self; 2], q: &[Self; 2]) -> std::cmp::Ordering {
418        use apfp::geometry::{Orientation, f64::Coord};
419        let orient = apfp::geometry::f64::orient2d_vec(
420          &Coord::new(p[0] as f64, p[1] as f64),
421          &Coord::new(vector[0] as f64, vector[1] as f64),
422          &Coord::new(q[0] as f64, q[1] as f64),
423        );
424        match orient {
425          Orientation::CounterClockwise => Ordering::Greater,
426          Orientation::Clockwise => Ordering::Less,
427          Orientation::CoLinear => Ordering::Equal,
428        }
429      }
430      // Uses apfp's orient2d_normal for fast arbitrary precision calculation.
431      fn cmp_perp_vector_slope(vector: &[Self;2], p: &[Self; 2], q: &[Self; 2]) -> std::cmp::Ordering {
432        use apfp::geometry::{Orientation, f64::Coord};
433        let orient = apfp::geometry::f64::orient2d_normal(
434          &Coord::new(p[0] as f64, p[1] as f64),
435          &Coord::new(vector[0] as f64, vector[1] as f64),
436          &Coord::new(q[0] as f64, q[1] as f64),
437        );
438        match orient {
439          Orientation::CounterClockwise => Ordering::Greater,
440          Orientation::Clockwise => Ordering::Less,
441          Orientation::CoLinear => Ordering::Equal,
442        }
443      }
444      fn incircle(
445        a: &[Self; 2],
446        b: &[Self; 2],
447        c: &[Self; 2],
448        d: &[Self; 2],
449      ) -> Ordering {
450        use apfp::geometry::{Orientation, f64::Coord};
451        let result = apfp::geometry::f64::incircle(
452          &Coord::new(a[0] as f64, a[1] as f64),
453          &Coord::new(b[0] as f64, b[1] as f64),
454          &Coord::new(c[0] as f64, c[1] as f64),
455          &Coord::new(d[0] as f64, d[1] as f64),
456        );
457        match result {
458          Orientation::CounterClockwise => Ordering::Greater,
459          Orientation::Clockwise => Ordering::Less,
460          Orientation::CoLinear => Ordering::Equal,
461        }
462      }
463    })*
464  };
465}

540    let long_p1: [i128; 2] = [p1[0] as i128, p1[1] as i128];
541    let long_p2: [i128; 2] = [p2[0] as i128, p2[1] as i128];
542    let long_q1: [i128; 2] = [q1[0] as i128, q1[1] as i128];
543    let long_q2: [i128; 2] = [q2[0] as i128, q2[1] as i128];
544    let long_result = approx_intersection_point_basic(0, long_p1, long_p2, long_q1, long_q2);
545    long_result.and_then(|[x, y]| Some([Self::try_from(x).ok()?, Self::try_from(y).ok()?]))
546  }
547  fn incircle(a: &[Self; 2], b: &[Self; 2], c: &[Self; 2], d: &[Self; 2]) -> Ordering {
548    use apfp::geometry::{Orientation, i64 as apfp_mod};
549    let result = apfp_mod::incircle(
550      &apfp_mod::Coord::new(a[0] as i64, a[1] as i64),
551      &apfp_mod::Coord::new(b[0] as i64, b[1] as i64),
552      &apfp_mod::Coord::new(c[0] as i64, c[1] as i64),
553      &apfp_mod::Coord::new(d[0] as i64, d[1] as i64),
554    );
555    match result {
556      Orientation::CounterClockwise => Ordering::Greater,
557      Orientation::Clockwise => Ordering::Less,
558      Orientation::CoLinear => Ordering::Equal,
559    }
560  }

574#[cfg(feature = "rug")]
575impl PolygonScalar for rug::Integer {
576  fn from_constant(val: i8) -> Self {
577    rug::Integer::from(val)
578  }
579  fn cmp_dist(p: &[Self; 2], q: &[Self; 2], r: &[Self; 2]) -> std::cmp::Ordering {
580    let [qx, qy] = q.clone();
581    let [px, py] = p.clone();
582    let pq_x = px - qx;
583    let pq_y = py - qy;
584    let pq_dist_squared: Self = pq_x.square() + pq_y.square();
585    let [rx, ry] = r.clone();
586    let [px, py] = p.clone();
587    let pr_x = px - rx;
588    let pr_y = py - ry;
589    let pr_dist_squared: Self = pr_x.square() + pr_y.square();
590    pq_dist_squared.cmp(&pr_dist_squared)
591  }
592
593  fn cmp_slope(p: &[Self; 2], q: &[Self; 2], r: &[Self; 2]) -> std::cmp::Ordering {
594    let [qx, qy] = q.clone();
595    let [rx, ry] = r.clone();
596    let ry_qy = ry - &q[1];
597    let qx_px = qx - &p[0];
598    let slope1 = ry_qy * qx_px;
599    let qy_py = qy - &p[1];
600    let rx_qx = rx - &q[0];
601    let slope2 = qy_py * rx_qx;
602    slope1.cmp(&slope2)
603  }
604  fn cmp_vector_slope(vector: &[Self; 2], p: &[Self; 2], q: &[Self; 2]) -> std::cmp::Ordering {
605    let new_x = rug::Integer::from(&p[0] + &vector[0]);
606    let new_y = rug::Integer::from(&p[1] + &vector[1]);
607    PolygonScalar::cmp_slope(p, &[new_x, new_y], q)
608  }
609  fn cmp_perp_vector_slope(vector: &[Self; 2], p: &[Self; 2], q: &[Self; 2]) -> std::cmp::Ordering {
610    let new_x = rug::Integer::from(&p[0] + &vector[1]);
611    let new_y = rug::Integer::from(&p[1] + &vector[0]);
612    PolygonScalar::cmp_slope(p, &[new_x, new_y], q)
613  }
614  fn incircle(a: &[Self; 2], b: &[Self; 2], c: &[Self; 2], d: &[Self; 2]) -> Ordering {
615    let mut ax = a[0].clone();
616    ax -= &d[0];
617    let mut ay = a[1].clone();
618    ay -= &d[1];
619    let mut bx = b[0].clone();
620    bx -= &d[0];
621    let mut by = b[1].clone();
622    by -= &d[1];
623    let mut cx = c[0].clone();
624    cx -= &d[0];
625    let mut cy = c[1].clone();
626    cy -= &d[1];
627
628    let a_len = ax.clone() * ax.clone() + ay.clone() * ay.clone();
629    let b_len = bx.clone() * bx.clone() + by.clone() * by.clone();
630    let c_len = cx.clone() * cx.clone() + cy.clone() * cy.clone();
631
632    let det = det3_generic(&ax, &ay, &a_len, &bx, &by, &b_len, &cx, &cy, &c_len);
633    det.cmp(&rug::Integer::new())
634  }
635}
