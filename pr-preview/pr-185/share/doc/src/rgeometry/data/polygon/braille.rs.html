4/// A pretty-printer for `Polygon<i8>` using Braille Unicode symbols.
5///
6/// Braille characters (U+2800 to U+28FF) encode 2×4 dot patterns, making them perfect
7/// for rendering small polygons as text-based graphics. Each Braille character represents
8/// a 2-pixel-wide by 4-pixel-tall cell.
9///
10/// # Braille Dot Numbering
11///
12/// Standard Braille numbering:
13/// ```text
14/// 1  4
15/// 2  5
16/// 3  6
17/// 7  8
18/// ```
19///
20/// # Examples
21///
22/// ```
23/// use rgeometry::data::{Point, Polygon};
24/// use rgeometry::data::polygon::BraillePrinter;
25///
26/// let triangle = Polygon::new(vec![
27///   Point::new([0, 0]),
28///   Point::new([10, 0]),
29///   Point::new([5, 8]),
30/// ]).unwrap();
31///
32/// let braille = BraillePrinter::new(&triangle);
33/// println!("{}", braille);
34/// ```
35pub struct BraillePrinter<'a> {
36  polygon: &'a Polygon<i8>,
37}
38
39impl<'a> BraillePrinter<'a> {
40  /// Create a new Braille printer for a polygon.
41  pub fn new(polygon: &'a Polygon<i8>) -> Self {
42    BraillePrinter { polygon }
43  }
44
45  /// Render the polygon to a string using Braille characters.
46  ///
47  /// # Examples
48  ///
49  /// ```
50  /// use rgeometry::data::{Point, Polygon};
51  /// use rgeometry::data::polygon::BraillePrinter;
52  ///
53  /// // Create a small triangle
54  /// let triangle = Polygon::new(vec![
55  ///   Point::new([0i8, 0]),
56  ///   Point::new([4, 0]),
57  ///   Point::new([2, 3]),
58  /// ]).unwrap();
59  ///
60  /// let printer = BraillePrinter::new(&triangle);
61  /// let output = format!("{}", printer);
62  ///
63  /// // The output should match this Braille pattern
64  /// assert_eq!(output, "⠐⡖⡖⠀\n⠀⠈⠀⠀\n");
65  /// ```
66  fn render(&self) -> String {
67    // Get bounding box
68    let (min, max) = self.polygon.bounding_box();
69    let min_x = min.x_coord();
70    let min_y = min.y_coord();
71    let max_x = max.x_coord();
72    let max_y = max.y_coord();
73
74    // Calculate dimensions with padding
75    let width = (max_x - min_x + 1) as usize + 2; // +2 for padding
76    let height = (max_y - min_y + 1) as usize + 2; // +2 for padding
77
78    // Create a pixel grid (false = empty, true = filled)
79    let mut grid = vec![vec![false; width]; height];
80
81    // Rasterize polygon edges using Bresenham's line algorithm
82    for edge in self.polygon.iter_boundary_edges() {
83      let x0 = (edge.src.x_coord() - min_x + 1) as i32;
84      let y0 = (edge.src.y_coord() - min_y + 1) as i32;
85      let x1 = (edge.dst.x_coord() - min_x + 1) as i32;
86      let y1 = (edge.dst.y_coord() - min_y + 1) as i32;
87
88      bresenham_line(x0, y0, x1, y1, |x, y| {
89        if x >= 0 && y >= 0 && (x as usize) < width && (y as usize) < height {
90          grid[y as usize][x as usize] = true;
91        }
92      });
93    }
94
95    // Convert grid to Braille characters
96    self.grid_to_braille(&grid)
97  }
98
99  /// Convert a pixel grid to Braille characters.
100  fn grid_to_braille(&self, grid: &[Vec<bool>]) -> String {
101    let height = grid.len();
102    let width = if height > 0 { grid[0].len() } else { 0 };
103
104    // Braille characters are 2 pixels wide and 4 pixels tall
105    let braille_rows = height.div_ceil(4);
106    let braille_cols = width.div_ceil(2);
107
108    let mut result = String::new();
109
110    for row in 0..braille_rows {
111      for col in 0..braille_cols {
112        let base_y = row * 4;
113        let base_x = col * 2;
114
115        // Map grid pixels to Braille dots
116        // Braille dot numbering (in Unicode order):
117        // 0: dot 1 (top-left)
118        // 1: dot 2 (middle-left)
119        // 2: dot 3 (bottom-left)
120        // 3: dot 4 (top-right)
121        // 4: dot 5 (middle-right)
122        // 5: dot 6 (bottom-right)
123        // 6: dot 7 (below bottom-left)
124        // 7: dot 8 (below bottom-right)
125
126        let mut dots = 0u8;
127
128        // Left column
129        if self.get_pixel(grid, base_x, base_y) {
130          dots |= 0b00000001; // dot 1
131        }
132        if self.get_pixel(grid, base_x, base_y + 1) {
133          dots |= 0b00000010; // dot 2
134        }
135        if self.get_pixel(grid, base_x, base_y + 2) {
136          dots |= 0b00000100; // dot 3
137        }
138        if self.get_pixel(grid, base_x, base_y + 3) {
139          dots |= 0b01000000; // dot 7
140        }
141
142        // Right column
143        if self.get_pixel(grid, base_x + 1, base_y) {
144          dots |= 0b00001000; // dot 4
145        }
146        if self.get_pixel(grid, base_x + 1, base_y + 1) {
147          dots |= 0b00010000; // dot 5
148        }
149        if self.get_pixel(grid, base_x + 1, base_y + 2) {
150          dots |= 0b00100000; // dot 6
151        }
152        if self.get_pixel(grid, base_x + 1, base_y + 3) {
153          dots |= 0b10000000; // dot 8
154        }
155
156        // Convert to Braille Unicode character
157        // U+2800 is the base Braille pattern (blank)
158        let braille_char = char::from_u32(0x2800 + dots as u32).unwrap_or('?');
159        result.push(braille_char);
160      }
161      result.push('\n');
162    }
163
164    result
165  }
166
167  /// Get a pixel from the grid, returning false if out of bounds.
168  fn get_pixel(&self, grid: &[Vec<bool>], x: usize, y: usize) -> bool {
169    grid
170      .get(y)
171      .and_then(|row| row.get(x))
172      .copied()
173      .unwrap_or(false)
174  }
175}

177impl<'a> fmt::Display for BraillePrinter<'a> {
178  fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
179    write!(f, "{}", self.render())
180  }
181}

183/// Bresenham's line drawing algorithm.
184///
185/// Calls the provided closure for each point along the line from (x0, y0) to (x1, y1).
186fn bresenham_line<F>(x0: i32, y0: i32, x1: i32, y1: i32, mut plot: F)
187where
188  F: FnMut(i32, i32),
189{
190  let dx = (x1 - x0).abs();
191  let dy = -(y1 - y0).abs();
192  let sx = if x0 < x1 { 1 } else { -1 };
193  let sy = if y0 < y1 { 1 } else { -1 };
194  let mut error = dx + dy;
195
196  let mut x = x0;
197  let mut y = y0;
198
199  loop {
200    plot(x, y);
201
202    if x == x1 && y == y1 {
203      break;
204    }
205
206    let e2 = 2 * error;
207
208    if e2 >= dy {
209      if x == x1 {
210        break;
211      }
212      error += dy;
213      x += sx;
214    }
215
216    if e2 <= dx {
217      if y == y1 {
218        break;
219      }
220      error += dx;
221      y += sy;
222    }
223  }
224}

231  #[test]
232  fn test_braille_triangle() {
233    let triangle = Polygon::new(vec![
234      Point::new([0, 0]),
235      Point::new([10, 0]),
236      Point::new([5, 8]),
237    ])
238    .unwrap();
239
240    let printer = BraillePrinter::new(&triangle);
241    let output = printer.render();
242
243    // The output should be non-empty and contain Braille characters
244    assert!(!output.is_empty());
245    assert!(
246      output
247        .chars()
248        .any(|c| ('\u{2800}'..='\u{28FF}').contains(&c))
249    );
250  }
