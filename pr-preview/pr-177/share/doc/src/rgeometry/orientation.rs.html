24  /// Determine the direction you have to turn if you walk from `p1`
25  /// to `p2` to `p3`.
26  ///
27  /// For fixed-precision types (i8,i16,i32,i64,etc), this function is
28  /// guaranteed to work for any input and never cause any arithmetic overflows.
29  ///
30  /// # Polymorphism
31  ///
32  /// This function works with both [Points](crate::data::Point) and [Vectors](Vector). You should prefer to
33  /// use [Point::orient](crate::data::Point::orient) when possible.
34  ///
35  /// # Examples
36  ///
37  /// ```rust
38  /// # use rgeometry::data::Point;
39  /// # use rgeometry::Orientation;
40  /// let p1 = Point::new([ 0, 0 ]);
41  /// let p2 = Point::new([ 0, 1 ]); // One unit above p1.
42  /// // (0,0) -> (0,1) -> (0,2) == Orientation::CoLinear
43  /// assert!(Orientation::new(&p1, &p2, &Point::new([ 0, 2 ])).is_colinear());
44  /// // (0,0) -> (0,1) -> (-1,2) == Orientation::CounterClockWise
45  /// assert!(Orientation::new(&p1, &p2, &Point::new([ -1, 2 ])).is_ccw());
46  /// // (0,0) -> (0,1) -> (1,2) == Orientation::ClockWise
47  /// assert!(Orientation::new(&p1, &p2, &Point::new([ 1, 2 ])).is_cw());
48  /// ```
49  ///
50  pub fn new<T>(p1: &[T; 2], p2: &[T; 2], p3: &[T; 2]) -> Orientation
51  where
52    T: PolygonScalar,
53  {
54    // raw_arr_turn(p, q, r)
55    match T::cmp_slope(p1, p2, p3) {
56      Ordering::Less => Orientation::ClockWise,
57      Ordering::Equal => Orientation::CoLinear,
58      Ordering::Greater => Orientation::CounterClockWise,
59    }
60  }

62  /// Locate `p2` in relation to the line determined by the point `p1` and the direction
63  /// vector.
64  ///
65  /// For fixed-precision types (i8,i16,i32,i64,etc), this function is
66  /// guaranteed to work for any input and never cause any arithmetic overflows.
67  ///
68  /// This function is identical to [`Orientation::new`]`(p1, p1+v, p2)` but will never
69  /// cause arithmetic overflows even if `p+v` would overflow.
70  ///
71  /// # Examples
72  ///
73  /// ```rust
74  /// # use rgeometry::data::{Vector,Point};
75  /// # use rgeometry::Orientation;
76  /// let v = Vector([ 1, 1 ]); // Vector pointing to the top-right corner.
77  /// let p1 = Point::new([ 5, 5 ]);
78  /// assert!(Orientation::along_vector(&p1, &v, &Point::new([ 6, 6 ])).is_colinear());
79  /// assert!(Orientation::along_vector(&p1, &v, &Point::new([ 7, 8 ])).is_ccw());
80  /// assert!(Orientation::along_vector(&p1, &v, &Point::new([ 8, 7 ])).is_cw());
81  /// ```
82  pub fn along_vector<T>(p1: &[T; 2], vector: &Vector<T, 2>, p2: &[T; 2]) -> Orientation
83  where
84    T: PolygonScalar,
85  {
86    match T::cmp_vector_slope(&vector.0, p1, p2) {
87      Ordering::Less => Orientation::ClockWise,
88      Ordering::Equal => Orientation::CoLinear,
89      Ordering::Greater => Orientation::CounterClockWise,
90    }
91  }

218  pub fn ccw_cmp_around_with<T>(
219    vector: &Vector<T, 2>,
220    p1: &[T; 2],
221    p2: &[T; 2],
222    p3: &[T; 2],
223  ) -> Ordering
224  where
225    T: PolygonScalar,
226  {
227    let aq = Orientation::along_vector(p1, vector, p2);
228    let ar = Orientation::along_vector(p1, vector, p3);
229    // let on_zero = |d: &[T; 2]| {
230    //   !((d[0] < p[0] && z[0].is_positive())
231    //     || (d[1] < p[1] && z[1].is_positive())
232    //     || (d[0] > p[0] && z[0].is_negative())
233    //     || (d[1] > p[1] && z[1].is_negative()))
234    // };
235    let on_zero = |d: &[T; 2]| match Orientation::along_perp_vector(p1, vector, d) {
236      CounterClockWise => false,
237      ClockWise => true,
238      CoLinear => true,
239    };
240    let cmp = || match Orientation::new(p1, p2, p3) {
241      CounterClockWise => Ordering::Less,
242      ClockWise => Ordering::Greater,
243      CoLinear => Ordering::Equal,
244    };
245    match (aq, ar) {
246      // Easy cases: Q and R are on either side of the line p->z:
247      (CounterClockWise, ClockWise) => Ordering::Less,
248      (ClockWise, CounterClockWise) => Ordering::Greater,
249      // A CoLinear point may be in front of p->z (0 degree angle) or behind
250      // it (180 degree angle). If the other point is clockwise, it must have an
251      // angle greater than 180 degrees and must therefore be greater than the
252      // colinear point.
253      (CoLinear, ClockWise) => Ordering::Less,
254      (ClockWise, CoLinear) => Ordering::Greater,
255
256      // if Q and R are on the same side of P->Z then the most clockwise point
257      // will have the smallest angle.
258      (CounterClockWise, CounterClockWise) => cmp(),
259      (ClockWise, ClockWise) => cmp(),
260
261      // CoLinear points have an angle of either 0 degrees or 180 degrees. on_zero
262      // can distinguish these two cases:
263      //    on_zero(p) => 0 degrees.
264      //   !on_zero(p) => 180 degrees.
265      (CounterClockWise, CoLinear) => {
266        if on_zero(p3) {
267          Ordering::Greater // angle(r) = 0 & 0 < angle(q) < 180. Thus: Q > R
268        } else {
269          Ordering::Less // angle(r) = 180 & 0 < angle(q) < 180. Thus: Q < R
270        }
271      }
272      (CoLinear, CounterClockWise) => {
273        if on_zero(p2) {
274          Ordering::Less
275        } else {
276          Ordering::Greater
277        }
278      }
279      (CoLinear, CoLinear) => match (on_zero(p2), on_zero(p3)) {
280        (true, true) => Ordering::Equal,
281        (false, false) => Ordering::Equal,
282        (true, false) => Ordering::Less,
283        (false, true) => Ordering::Greater,
284      },
285    }
286  }

364  /// Compare angles around an origin point where:
365  /// - The reference direction is an edge's outward normal (ref_edge_start → ref_edge_end)
366  /// - One comparison point is another edge's outward normal (edge_start → edge_end)
367  /// - The other comparison point is an explicit direction vector
368  ///
369  /// This is a specialized version of `ccw_cmp_around_with` designed for the
370  /// `extreme_in_direction` algorithm that avoids overflow when computing edge normals.
371  ///
372  /// Returns the ordering of the edge normal's angle compared to the direction's angle,
373  /// both measured from the reference normal as the zero angle.
374  ///
375  /// For fixed-precision types (i8,i16,i32,i64,etc), this function is
376  /// guaranteed to work for any input and never cause any arithmetic overflows.
377  pub fn ccw_cmp_around_with_edge_normal_vs_direction<T>(
378    ref_edge_start: &[T; 2],
379    ref_edge_end: &[T; 2],
380    p1: &[T; 2],
381    edge_start: &[T; 2],
382    edge_end: &[T; 2],
383    direction: &[T; 2],
384  ) -> Ordering
385  where
386    T: PolygonScalar,
387  {
388    // p2 = edge normal (implicitly defined by edge_start, edge_end)
389    // p3 = direction (explicit)
390
391    // aq = orientation of edge_normal relative to reference_normal line through p1
392    let aq = Orientation::along_ref_edge_normal_edge(
393      ref_edge_start,
394      ref_edge_end,
395      edge_start,
396      edge_end,
397    );
398    // ar = orientation of direction relative to reference_normal line through p1
399    let ar = Orientation::along_edge_normal_point(p1, ref_edge_start, ref_edge_end, direction);
400
401    // on_zero_edge = is edge_normal at angle 0 (in front) or 180 (behind)?
402    let on_zero_edge = || match Orientation::along_perp_ref_edge_normal_edge(
403      ref_edge_start,
404      ref_edge_end,
405      edge_start,
406      edge_end,
407    ) {
408      CounterClockWise => false,
409      ClockWise => true,
410      CoLinear => true,
411    };
412
413    // on_zero_dir = is direction at angle 0 (in front) or 180 (behind)?
414    let on_zero_dir = || match Orientation::along_perp_edge_normal_point(
415      p1,
416      ref_edge_start,
417      ref_edge_end,
418      direction,
419    ) {
420      CounterClockWise => false,
421      ClockWise => true,
422      CoLinear => true,
423    };
424
425    // cmp = direct orientation comparison of edge_normal vs direction around p1
426    let cmp = || {
427      // We need Orientation::new(p1, edge_normal, direction)
428      // This returns CCW if direction is CCW (left) of p1→edge_normal,
429      // which means edge_normal has a smaller angle than direction.
430      //
431      // cmp_edge_normal_vs_direction returns sign(edge_normal × direction):
432      // - Greater (positive): direction is CCW from edge_normal → edge_normal < direction → Less
433      // - Less (negative): direction is CW from edge_normal → edge_normal > direction → Greater
434      // - Equal: collinear
435      //
436      // This mapping matches the original ccw_cmp_around_with logic where:
437      // - CCW → Less (q comes before r)
438      // - CW → Greater (q comes after r)
439      match Orientation::cmp_edge_normal_vs_direction(edge_start, edge_end, direction) {
440        Ordering::Greater => Ordering::Less, // edge_normal × direction > 0 → edge_normal < direction
441        Ordering::Less => Ordering::Greater, // edge_normal × direction < 0 → edge_normal > direction
442        Ordering::Equal => Ordering::Equal,
443      }
444    };
445
446    match (aq, ar) {
447      // Easy cases: Q and R are on either side of the line p->z:
448      (CounterClockWise, ClockWise) => Ordering::Less,
449      (ClockWise, CounterClockWise) => Ordering::Greater,
450      // A CoLinear point may be in front of p->z (0 degree angle) or behind
451      // it (180 degree angle). If the other point is clockwise, it must have an
452      // angle greater than 180 degrees and must therefore be greater than the
453      // colinear point.
454      (CoLinear, ClockWise) => Ordering::Less,
455      (ClockWise, CoLinear) => Ordering::Greater,
456
457      // if Q and R are on the same side of P->Z then the most clockwise point
458      // will have the smallest angle.
459      (CounterClockWise, CounterClockWise) => cmp(),
460      (ClockWise, ClockWise) => cmp(),
461
462      // CoLinear points have an angle of either 0 degrees or 180 degrees.
463      (CounterClockWise, CoLinear) => {
464        if on_zero_dir() {
465          Ordering::Greater
466        } else {
467          Ordering::Less
468        }
469      }
470      (CoLinear, CounterClockWise) => {
471        if on_zero_edge() {
472          Ordering::Less
473        } else {
474          Ordering::Greater
475        }
476      }
477      (CoLinear, CoLinear) => match (on_zero_edge(), on_zero_dir()) {
478        (true, true) => Ordering::Equal,
479        (false, false) => Ordering::Equal,
480        (true, false) => Ordering::Less,
481        (false, true) => Ordering::Greater,
482      },
483    }
484  }

486  /// Orientation of a point relative to a line defined by an edge's outward normal.
487  /// This is `along_edge_normal` but for an explicit point instead of another edge.
488  fn along_edge_normal_point<T>(
489    p1: &[T; 2],
490    ref_edge_start: &[T; 2],
491    ref_edge_end: &[T; 2],
492    point: &[T; 2],
493  ) -> Orientation
494  where
495    T: PolygonScalar,
496  {
497    // The edge normal is (ref_edge_end[1] - ref_edge_start[1], ref_edge_start[0] - ref_edge_end[0])
498    // along_vector(p1, normal, point) computes orientation of point relative to line p1→(p1+normal)
499    // This is: sign((p1 - point) × normal) = sign((p1.x - point.x) * normal.y - (p1.y - point.y) * normal.x)
500    // = sign((p1.x - point.x) * (ref_edge_start[0] - ref_edge_end[0]) - (p1.y - point.y) * (ref_edge_end[1] - ref_edge_start[1]))
501    match T::cmp_edge_normal_slope(ref_edge_start, ref_edge_end, p1, point) {
502      Ordering::Less => Orientation::ClockWise,
503      Ordering::Equal => Orientation::CoLinear,
504      Ordering::Greater => Orientation::CounterClockWise,
505    }
506  }

555// https://arxiv.org/abs/math/9410209
556// Simulation of Simplicity.
557// Break ties (ie colinear orientations) in an arbitrary but consistent way.
558impl SoS {
559  // p: Point::new([a, 2^a])
560  // q: Point::new([b, 2^b])
561  // r: Point::new([c, 2^c])
562  // new(a,b,c) == Orientation::new(p, q, r)
563  pub fn new(a: u32, b: u32, c: u32) -> SoS {
564    assert_ne!(a, b);
565    assert_ne!(b, c);
566    assert_ne!(c, a);
567
568    // Combinations:
569    //               a<b a<c c<b
570    // b a c => CW    _   X   _
571    // c b a => CW    _   _   X
572    // a c b => CW    X   X   X
573    // b c a => CCW   _   _   _
574    // c a b => CCW   X   _   X
575    // a b c => CCW   X   X   _
576
577    let ab = a < b;
578    let ac = a < c;
579    let cb = c < b;
580    if ab ^ ac ^ cb {
581      SoS::ClockWise
582    } else {
583      SoS::CounterClockWise
584    }
585    // if a < b {
586    //   if a < c && c < b {
587    //     SoS::ClockWise // a c b
588    //   } else {
589    //     SoS::CounterClockWise // a b c, c a b
590    //   }
591    // } else if b < c && c < a {
592    //   SoS::CounterClockWise // b c a
593    // } else {
594    //   SoS::ClockWise // b a c, c b a
595    // }
596  }
597
598  pub fn orient(self) -> Orientation {
599    match self {
600      SoS::CounterClockWise => Orientation::CounterClockWise,
601      SoS::ClockWise => Orientation::ClockWise,
602    }
603  }
604
605  #[must_use]
606  pub fn reverse(self) -> SoS {
607    match self {
608      SoS::CounterClockWise => SoS::ClockWise,
609      SoS::ClockWise => SoS::CounterClockWise,
610    }
611  }
612}
