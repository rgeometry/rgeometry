24  /// Determine the direction you have to turn if you walk from `p1`
25  /// to `p2` to `p3`.
26  ///
27  /// For fixed-precision types (i8,i16,i32,i64,etc), this function is
28  /// guaranteed to work for any input and never cause any arithmetic overflows.
29  ///
30  /// # Polymorphism
31  ///
32  /// This function works with both [Points](crate::data::Point) and [Vectors](Vector). You should prefer to
33  /// use [Point::orient](crate::data::Point::orient) when possible.
34  ///
35  /// # Examples
36  ///
37  /// ```rust
38  /// # use rgeometry::data::Point;
39  /// # use rgeometry::Orientation;
40  /// let p1 = Point::new([ 0, 0 ]);
41  /// let p2 = Point::new([ 0, 1 ]); // One unit above p1.
42  /// // (0,0) -> (0,1) -> (0,2) == Orientation::CoLinear
43  /// assert!(Orientation::new(&p1, &p2, &Point::new([ 0, 2 ])).is_colinear());
44  /// // (0,0) -> (0,1) -> (-1,2) == Orientation::CounterClockWise
45  /// assert!(Orientation::new(&p1, &p2, &Point::new([ -1, 2 ])).is_ccw());
46  /// // (0,0) -> (0,1) -> (1,2) == Orientation::ClockWise
47  /// assert!(Orientation::new(&p1, &p2, &Point::new([ 1, 2 ])).is_cw());
48  /// ```
49  ///
50  pub fn new<T>(p1: &[T; 2], p2: &[T; 2], p3: &[T; 2]) -> Orientation
51  where
52    T: PolygonScalar,
53  {
54    // raw_arr_turn(p, q, r)
55    match T::cmp_slope(p1, p2, p3) {
56      Ordering::Less => Orientation::ClockWise,
57      Ordering::Equal => Orientation::CoLinear,
58      Ordering::Greater => Orientation::CounterClockWise,
59    }
60  }

62  /// Locate `p2` in relation to the line determined by the point `p1` and the direction
63  /// vector.
64  ///
65  /// For fixed-precision types (i8,i16,i32,i64,etc), this function is
66  /// guaranteed to work for any input and never cause any arithmetic overflows.
67  ///
68  /// This function is identical to [`Orientation::new`]`(p1, p1+v, p2)` but will never
69  /// cause arithmetic overflows even if `p+v` would overflow.
70  ///
71  /// # Examples
72  ///
73  /// ```rust
74  /// # use rgeometry::data::{Vector,Point};
75  /// # use rgeometry::Orientation;
76  /// let v = Vector([ 1, 1 ]); // Vector pointing to the top-right corner.
77  /// let p1 = Point::new([ 5, 5 ]);
78  /// assert!(Orientation::along_vector(&p1, &v, &Point::new([ 6, 6 ])).is_colinear());
79  /// assert!(Orientation::along_vector(&p1, &v, &Point::new([ 7, 8 ])).is_ccw());
80  /// assert!(Orientation::along_vector(&p1, &v, &Point::new([ 8, 7 ])).is_cw());
81  /// ```
82  pub fn along_vector<T>(p1: &[T; 2], vector: &Vector<T, 2>, p2: &[T; 2]) -> Orientation
83  where
84    T: PolygonScalar,
85  {
86    match T::cmp_vector_slope(&vector.0, p1, p2) {
87      Ordering::Less => Orientation::ClockWise,
88      Ordering::Equal => Orientation::CoLinear,
89      Ordering::Greater => Orientation::CounterClockWise,
90    }
91  }

149  /// Check if an edge's normal is on the positive side of the perpendicular to a reference edge's normal.
150  /// This is used to determine if an edge_normal has angle 0° (in front) or 180° (behind) relative
151  /// to the reference normal.
152  fn along_perp_ref_edge_normal_edge<T>(
153    _p1: &[T; 2],
154    ref_edge_start: &[T; 2],
155    ref_edge_end: &[T; 2],
156    edge_start: &[T; 2],
157    edge_end: &[T; 2],
158  ) -> Orientation
159  where
160    T: PolygonScalar,
161  {
162    // We need to determine if edge_normal is at angle 0° (same direction as ref_normal)
163    // or at angle 180° (opposite direction).
164    //
165    // This is equivalent to checking if edge_normal · ref_normal > 0:
166    // - Positive: same direction (0°) → on_zero = true
167    // - Negative: opposite direction (180°) → on_zero = false
168    // - Zero: perpendicular (90° or 270°) → treat as on_zero = true (consistent with original)
169    //
170    // Since normal1 · normal2 = edge1 · edge2 (dot product of normals equals dot product of edges),
171    // we use cmp_edge_normals_dot which computes this without overflow.
172    match T::cmp_edge_normals_dot(ref_edge_start, ref_edge_end, edge_start, edge_end) {
173      Ordering::Greater => Orientation::ClockWise, // Same direction → on_zero = true
174      Ordering::Less => Orientation::CounterClockWise, // Opposite direction → on_zero = false
175      Ordering::Equal => Orientation::CoLinear,    // Perpendicular → on_zero = true
176    }
177  }

231  pub fn ccw_cmp_around_with<T>(
232    vector: &Vector<T, 2>,
233    p1: &[T; 2],
234    p2: &[T; 2],
235    p3: &[T; 2],
236  ) -> Ordering
237  where
238    T: PolygonScalar,
239  {
240    let aq = Orientation::along_vector(p1, vector, p2);
241    let ar = Orientation::along_vector(p1, vector, p3);
242    // let on_zero = |d: &[T; 2]| {
243    //   !((d[0] < p[0] && z[0].is_positive())
244    //     || (d[1] < p[1] && z[1].is_positive())
245    //     || (d[0] > p[0] && z[0].is_negative())
246    //     || (d[1] > p[1] && z[1].is_negative()))
247    // };
248    let on_zero = |d: &[T; 2]| match Orientation::along_perp_vector(p1, vector, d) {
249      CounterClockWise => false,
250      ClockWise => true,
251      CoLinear => true,
252    };
253    let cmp = || match Orientation::new(p1, p2, p3) {
254      CounterClockWise => Ordering::Less,
255      ClockWise => Ordering::Greater,
256      CoLinear => Ordering::Equal,
257    };
258    match (aq, ar) {
259      // Easy cases: Q and R are on either side of the line p->z:
260      (CounterClockWise, ClockWise) => Ordering::Less,
261      (ClockWise, CounterClockWise) => Ordering::Greater,
262      // A CoLinear point may be in front of p->z (0 degree angle) or behind
263      // it (180 degree angle). If the other point is clockwise, it must have an
264      // angle greater than 180 degrees and must therefore be greater than the
265      // colinear point.
266      (CoLinear, ClockWise) => Ordering::Less,
267      (ClockWise, CoLinear) => Ordering::Greater,
268
269      // if Q and R are on the same side of P->Z then the most clockwise point
270      // will have the smallest angle.
271      (CounterClockWise, CounterClockWise) => cmp(),
272      (ClockWise, ClockWise) => cmp(),
273
274      // CoLinear points have an angle of either 0 degrees or 180 degrees. on_zero
275      // can distinguish these two cases:
276      //    on_zero(p) => 0 degrees.
277      //   !on_zero(p) => 180 degrees.
278      (CounterClockWise, CoLinear) => {
279        if on_zero(p3) {
280          Ordering::Greater // angle(r) = 0 & 0 < angle(q) < 180. Thus: Q > R
281        } else {
282          Ordering::Less // angle(r) = 180 & 0 < angle(q) < 180. Thus: Q < R
283        }
284      }
285      (CoLinear, CounterClockWise) => {
286        if on_zero(p2) {
287          Ordering::Less
288        } else {
289          Ordering::Greater
290        }
291      }
292      (CoLinear, CoLinear) => match (on_zero(p2), on_zero(p3)) {
293        (true, true) => Ordering::Equal,
294        (false, false) => Ordering::Equal,
295        (true, false) => Ordering::Less,
296        (false, true) => Ordering::Greater,
297      },
298    }
299  }

377  /// Compare angles around an origin point where:
378  /// - The reference direction is an edge's outward normal (ref_edge_start → ref_edge_end)
379  /// - One comparison point is another edge's outward normal (edge_start → edge_end)
380  /// - The other comparison point is an explicit direction vector
381  ///
382  /// This is a specialized version of `ccw_cmp_around_with` designed for the
383  /// `extreme_in_direction` algorithm that avoids overflow when computing edge normals.
384  ///
385  /// Returns the ordering of the edge normal's angle compared to the direction's angle,
386  /// both measured from the reference normal as the zero angle.
387  ///
388  /// For fixed-precision types (i8,i16,i32,i64,etc), this function is
389  /// guaranteed to work for any input and never cause any arithmetic overflows.
390  pub fn ccw_cmp_around_with_edge_normal_vs_direction<T>(
391    ref_edge_start: &[T; 2],
392    ref_edge_end: &[T; 2],
393    p1: &[T; 2],
394    edge_start: &[T; 2],
395    edge_end: &[T; 2],
396    direction: &[T; 2],
397  ) -> Ordering
398  where
399    T: PolygonScalar,
400  {
401    // p2 = edge normal (implicitly defined by edge_start, edge_end)
402    // p3 = direction (explicit)
403
404    // aq = orientation of edge_normal relative to reference_normal line through p1
405    let aq = Orientation::along_ref_edge_normal_edge(
406      p1,
407      ref_edge_start,
408      ref_edge_end,
409      edge_start,
410      edge_end,
411    );
412    // ar = orientation of direction relative to reference_normal line through p1
413    let ar = Orientation::along_edge_normal_point(p1, ref_edge_start, ref_edge_end, direction);
414
415    // on_zero_edge = is edge_normal at angle 0 (in front) or 180 (behind)?
416    let on_zero_edge = || match Orientation::along_perp_ref_edge_normal_edge(
417      p1,
418      ref_edge_start,
419      ref_edge_end,
420      edge_start,
421      edge_end,
422    ) {
423      CounterClockWise => false,
424      ClockWise => true,
425      CoLinear => true,
426    };
427
428    // on_zero_dir = is direction at angle 0 (in front) or 180 (behind)?
429    let on_zero_dir = || match Orientation::along_perp_edge_normal_point(
430      p1,
431      ref_edge_start,
432      ref_edge_end,
433      direction,
434    ) {
435      CounterClockWise => false,
436      ClockWise => true,
437      CoLinear => true,
438    };
439
440    // cmp = direct orientation comparison of edge_normal vs direction around p1
441    let cmp = || {
442      // We need Orientation::new(p1, edge_normal, direction)
443      // This returns CCW if direction is CCW (left) of p1→edge_normal,
444      // which means edge_normal has a smaller angle than direction.
445      //
446      // cmp_edge_normal_vs_direction returns sign(edge_normal × direction):
447      // - Greater (positive): direction is CCW from edge_normal → edge_normal < direction → Less
448      // - Less (negative): direction is CW from edge_normal → edge_normal > direction → Greater
449      // - Equal: collinear
450      //
451      // This mapping matches the original ccw_cmp_around_with logic where:
452      // - CCW → Less (q comes before r)
453      // - CW → Greater (q comes after r)
454      match Orientation::cmp_edge_normal_vs_direction(edge_start, edge_end, direction) {
455        Ordering::Greater => Ordering::Less, // edge_normal × direction > 0 → edge_normal < direction
456        Ordering::Less => Ordering::Greater, // edge_normal × direction < 0 → edge_normal > direction
457        Ordering::Equal => Ordering::Equal,
458      }
459    };
460
461    match (aq, ar) {
462      // Easy cases: Q and R are on either side of the line p->z:
463      (CounterClockWise, ClockWise) => Ordering::Less,
464      (ClockWise, CounterClockWise) => Ordering::Greater,
465      // A CoLinear point may be in front of p->z (0 degree angle) or behind
466      // it (180 degree angle). If the other point is clockwise, it must have an
467      // angle greater than 180 degrees and must therefore be greater than the
468      // colinear point.
469      (CoLinear, ClockWise) => Ordering::Less,
470      (ClockWise, CoLinear) => Ordering::Greater,
471
472      // if Q and R are on the same side of P->Z then the most clockwise point
473      // will have the smallest angle.
474      (CounterClockWise, CounterClockWise) => cmp(),
475      (ClockWise, ClockWise) => cmp(),
476
477      // CoLinear points have an angle of either 0 degrees or 180 degrees.
478      (CounterClockWise, CoLinear) => {
479        if on_zero_dir() {
480          Ordering::Greater
481        } else {
482          Ordering::Less
483        }
484      }
485      (CoLinear, CounterClockWise) => {
486        if on_zero_edge() {
487          Ordering::Less
488        } else {
489          Ordering::Greater
490        }
491      }
492      (CoLinear, CoLinear) => match (on_zero_edge(), on_zero_dir()) {
493        (true, true) => Ordering::Equal,
494        (false, false) => Ordering::Equal,
495        (true, false) => Ordering::Less,
496        (false, true) => Ordering::Greater,
497      },
498    }
499  }

501  /// Orientation of a point relative to a line defined by an edge's outward normal.
502  /// This is `along_edge_normal` but for an explicit point instead of another edge.
503  fn along_edge_normal_point<T>(
504    p1: &[T; 2],
505    ref_edge_start: &[T; 2],
506    ref_edge_end: &[T; 2],
507    point: &[T; 2],
508  ) -> Orientation
509  where
510    T: PolygonScalar,
511  {
512    // The edge normal is (ref_edge_end[1] - ref_edge_start[1], ref_edge_start[0] - ref_edge_end[0])
513    // along_vector(p1, normal, point) computes orientation of point relative to line p1→(p1+normal)
514    // This is: sign((p1 - point) × normal) = sign((p1.x - point.x) * normal.y - (p1.y - point.y) * normal.x)
515    // = sign((p1.x - point.x) * (ref_edge_start[0] - ref_edge_end[0]) - (p1.y - point.y) * (ref_edge_end[1] - ref_edge_start[1]))
516    match T::cmp_edge_normal_slope(ref_edge_start, ref_edge_end, p1, point) {
517      Ordering::Less => Orientation::ClockWise,
518      Ordering::Equal => Orientation::CoLinear,
519      Ordering::Greater => Orientation::CounterClockWise,
520    }
521  }

540  /// Orientation of an edge's outward normal relative to another edge's normal line.
541  /// This computes whether edge_normal is CCW/CW from the line through p1 in direction ref_normal.
542  fn along_ref_edge_normal_edge<T>(
543    _p1: &[T; 2],
544    ref_edge_start: &[T; 2],
545    ref_edge_end: &[T; 2],
546    edge_start: &[T; 2],
547    edge_end: &[T; 2],
548  ) -> Orientation
549  where
550    T: PolygonScalar,
551  {
552    // We want: sign((p1 - edge_normal) × ref_normal)
553    // where edge_normal = (edge_end[1] - edge_start[1], edge_start[0] - edge_end[0])
554    // and ref_normal = (ref_edge_end[1] - ref_edge_start[1], ref_edge_start[0] - ref_edge_end[0])
555    //
556    // But p1 is the origin, so p1 - edge_normal = -edge_normal.
557    // So we need: sign(-edge_normal × ref_normal) = -sign(edge_normal × ref_normal)
558    //
559    // edge_normal × ref_normal = edge_normal.x * ref_normal.y - edge_normal.y * ref_normal.x
560    // = (edge_end[1] - edge_start[1]) * (ref_edge_start[0] - ref_edge_end[0])
561    //   - (edge_start[0] - edge_end[0]) * (ref_edge_end[1] - ref_edge_start[1])
562
563    // Let's use the cmp_edge_normal_slope formulation:
564    // cmp_edge_normal_slope(ref_start, ref_end, p1, q) computes orientation relative to ref's normal
565    // But here we need to compare edge_normal as a point (vector from origin).
566    // edge_normal as a "point" from origin is just the normal itself.
567
568    // Actually, we need to think about this differently.
569    // along_vector(p1, v, point) computes whether point is CCW/CW from line p1→(p1+v).
570    // Here p1 = origin = [0,0], v = ref_normal, point = edge_normal.
571    //
572    // The computation is: sign((p1 - point) × v) = sign(-point × v) = sign(v × point)
573    // = sign(ref_normal × edge_normal)
574    // = sign(ref_normal.x * edge_normal.y - ref_normal.y * edge_normal.x)
575
576    // ref_normal = (ref_edge_end[1] - ref_edge_start[1], ref_edge_start[0] - ref_edge_end[0])
577    // edge_normal = (edge_end[1] - edge_start[1], edge_start[0] - edge_end[0])
578
579    // ref_normal × edge_normal
580    // = (ref_edge_end[1] - ref_edge_start[1]) * (edge_start[0] - edge_end[0])
581    //   - (ref_edge_start[0] - ref_edge_end[0]) * (edge_end[1] - edge_start[1])
582
583    // This needs a new primitive. For now, let me use a workaround using existing primitives.
584    // We can use cmp_edge_normal_slope which computes:
585    // sign((p[0] - q[0]) * (e0 - f0) - (p[1] - q[1]) * (f1 - e1))
586    // where (e0,e1) = edge_start, (f0,f1) = edge_end
587
588    // If p = edge_normal and q = origin = [0,0]:
589    // We'd get: sign(edge_normal.x * (ref_start[0] - ref_end[0]) - edge_normal.y * (ref_end[1] - ref_start[1]))
590    // But we can't compute edge_normal explicitly!
591
592    // Alternative approach: use cmp_slope with carefully constructed points.
593    // cmp_slope(p, q, r) computes sign((r[1] - q[1]) * (q[0] - p[0]) - (q[1] - p[1]) * (r[0] - q[0]))
594
595    // Let me try a different formulation. The cross product of two edge normals:
596    // n1 = (dy1, -dx1) where (dx1, dy1) = ref_edge_end - ref_edge_start
597    // n2 = (dy2, -dx2) where (dx2, dy2) = edge_end - edge_start
598    //
599    // n1 × n2 = dy1 * (-dx2) - (-dx1) * dy2 = -dy1*dx2 + dx1*dy2 = dx1*dy2 - dy1*dx2
600    //
601    // This is the negative of the cross product of the edge vectors!
602    // edge1 × edge2 = dx1*dy2 - dy1*dx2 (same!)
603    //
604    // Wait, that's wrong. Let me redo:
605    // n1 × n2 = n1.x * n2.y - n1.y * n2.x
606    //         = dy1 * (-dx2) - (-dx1) * dy2
607    //         = -dy1*dx2 + dx1*dy2
608    //         = dx1*dy2 - dy1*dx2
609
610    // Meanwhile, edge1 × edge2 = dx1*dy2 - dy1*dx2 (same!)
611
612    // So the cross product of two normals equals the cross product of the corresponding edges!
613    // This means we can use orient2d on the edge endpoints.
614
615    // Specifically: sign(n1 × n2) = sign(edge1 × edge2)
616    // = sign((ref_edge_end - ref_edge_start) × (edge_end - edge_start))
617    // = sign(orient2d(ref_edge_start, ref_edge_end, edge_end) - orient2d(ref_edge_start, ref_edge_end, edge_start))
618    // ... this is getting complicated.
619
620    // Actually, for edge1 × edge2 with edge_i = end_i - start_i:
621    // This equals orient2d(start1, end1, end2) scaled somehow... but it's not quite right because
622    // the edges don't share a point.
623
624    // Let me use a different approach. We have:
625    // sign(ref_normal × edge_normal) where both are at origin.
626    // = sign(ref_normal.x * edge_normal.y - ref_normal.y * edge_normal.x)
627    // = sign((ref_end[1] - ref_start[1]) * (edge_start[0] - edge_end[0])
628    //        - (ref_start[0] - ref_end[0]) * (edge_end[1] - edge_start[1]))
629    //
630    // Expanding:
631    // = sign((ref_end[1] - ref_start[1]) * (edge_start[0] - edge_end[0])
632    //        + (ref_end[0] - ref_start[0]) * (edge_end[1] - edge_start[1]))
633
634    // This needs a new primitive. Let me add it.
635    match T::cmp_edge_normals_cross(ref_edge_start, ref_edge_end, edge_start, edge_end) {
636      Ordering::Greater => Orientation::CounterClockWise,
637      Ordering::Less => Orientation::ClockWise,
638      Ordering::Equal => Orientation::CoLinear,
639    }
640  }
641
642  /// Compare an edge normal against a direction vector.
643  /// Returns the cross product sign: sign(edge_normal × direction).
644  fn cmp_edge_normal_vs_direction<T>(
645    edge_start: &[T; 2],
646    edge_end: &[T; 2],
647    direction: &[T; 2],
648  ) -> Ordering
649  where
650    T: PolygonScalar,
651  {
652    // edge_normal = (edge_end[1] - edge_start[1], edge_start[0] - edge_end[0])
653    // edge_normal × direction = edge_normal.x * direction.y - edge_normal.y * direction.x
654    // = (edge_end[1] - edge_start[1]) * direction[1] - (edge_start[0] - edge_end[0]) * direction[0]
655    // = (edge_end[1] - edge_start[1]) * direction[1] + (edge_end[0] - edge_start[0]) * direction[0]
656    // This is the dot product of (edge_end - edge_start) and direction!
657
658    // Actually wait, let me redo this:
659    // edge_normal × direction = n.x * d.y - n.y * d.x
660    // n = (e_end.y - e_start.y, e_start.x - e_end.x)
661    // = (e_end.y - e_start.y) * d.y - (e_start.x - e_end.x) * d.x
662    // = (e_end.y - e_start.y) * d.y + (e_end.x - e_start.x) * d.x
663
664    // This IS the dot product of edge_vector and direction!
665    // (edge_end - edge_start) · direction
666
667    T::cmp_edge_normal_cross_direction(edge_start, edge_end, direction)
668  }
669}

671// https://arxiv.org/abs/math/9410209
672// Simulation of Simplicity.
673// Break ties (ie colinear orientations) in an arbitrary but consistent way.
674impl SoS {
675  // p: Point::new([a, 2^a])
676  // q: Point::new([b, 2^b])
677  // r: Point::new([c, 2^c])
678  // new(a,b,c) == Orientation::new(p, q, r)
679  pub fn new(a: u32, b: u32, c: u32) -> SoS {
680    assert_ne!(a, b);
681    assert_ne!(b, c);
682    assert_ne!(c, a);
683
684    // Combinations:
685    //               a<b a<c c<b
686    // b a c => CW    _   X   _
687    // c b a => CW    _   _   X
688    // a c b => CW    X   X   X
689    // b c a => CCW   _   _   _
690    // c a b => CCW   X   _   X
691    // a b c => CCW   X   X   _
692
693    let ab = a < b;
694    let ac = a < c;
695    let cb = c < b;
696    if ab ^ ac ^ cb {
697      SoS::ClockWise
698    } else {
699      SoS::CounterClockWise
700    }
701    // if a < b {
702    //   if a < c && c < b {
703    //     SoS::ClockWise // a c b
704    //   } else {
705    //     SoS::CounterClockWise // a b c, c a b
706    //   }
707    // } else if b < c && c < a {
708    //   SoS::CounterClockWise // b c a
709    // } else {
710    //   SoS::ClockWise // b a c, c b a
711    // }
712  }
713
714  pub fn orient(self) -> Orientation {
715    match self {
716      SoS::CounterClockWise => Orientation::CounterClockWise,
717      SoS::ClockWise => Orientation::ClockWise,
718    }
719  }
720
721  #[must_use]
722  pub fn reverse(self) -> SoS {
723    match self {
724      SoS::CounterClockWise => SoS::ClockWise,
725      SoS::ClockWise => SoS::CounterClockWise,
726    }
727  }
728}
